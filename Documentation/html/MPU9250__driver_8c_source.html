34uint8_t MPU_begin(I2C_HandleTypeDef *I2Cx, uint8_t addr, uint8_t aScale, uint8_t gScale, float tau, float dt)
35{
36 // Save values
37 _addr = addr << 1;
38 _tau = tau;
39 _dt = dt;
40
41 // Initialize variables
42 uint8_t check;
43 uint8_t select;
44
45 // Confirm device
46 HAL_I2C_Mem_Read(I2Cx, _addr, WHO_AM_I, 1, &check, 1, I2C_TIMOUT_MS);
47
48 // TODO: If 9250 or 6050 fails could it trigger the opposite check???
49 if ((check == WHO_AM_I_9250_ANS) || (check == WHO_AM_I_6050_ANS))
50 {
51 // Startup / reset the sensor
52 select = 0x00;
53 HAL_I2C_Mem_Write(I2Cx, _addr, PWR_MGMT_1, 1, &select, 1, I2C_TIMOUT_MS);
54
55 // Set the full scale ranges
56 MPU_writeAccFullScaleRange(I2Cx, aScale);
57 MPU_writeGyroFullScaleRange(I2Cx, gScale);
58
59 return 1;
60 }
61 else

63 return 0;
64 }

147void MPU_readRawData(I2C_HandleTypeDef *I2Cx)
148{
149 // Init buffer
150 uint8_t buf[14];
151
152 // Subroutine for reading the raw data
153 HAL_I2C_Mem_Read(I2Cx, _addr, ACCEL_XOUT_H, 1, buf, 14, I2C_TIMOUT_MS);
154
155 // Bit shift the data
156 rawData.ax = buf[0] << 8 | buf[1];
157 rawData.ay = buf[2] << 8 | buf[3];
158 rawData.az = buf[4] << 8 | buf[5];
159 // temperature = buf[6] << 8 | buf[7];
160 rawData.gx = buf[8] << 8 | buf[9];
161 rawData.gy = buf[10] << 8 | buf[11];
162 rawData.gz = buf[12] << 8 | buf[13];
163}

168void MPU_calibrateGyro(I2C_HandleTypeDef *I2Cx, uint16_t numCalPoints)
169{
170 // Init
171 int32_t x = 0;
172 int32_t y = 0;
173 int32_t z = 0;
174
175 // Zero guard
176 if (numCalPoints == 0)
177 {
178 numCalPoints = 1;
179 }
180
181 // Save specified number of points
182 for (uint16_t ii = 0; ii < numCalPoints; ii++)
183 {
184 MPU_readRawData(I2Cx);
185 x += rawData.gx;
186 y += rawData.gy;
187 z += rawData.gz;
188 HAL_Delay(3);
189 }
190
191 // Average the saved data points to find the gyroscope offset
192 gyroCal.x = (float)x / (float)numCalPoints;
193 gyroCal.y = (float)y / (float)numCalPoints;
194 gyroCal.z = (float)z / (float)numCalPoints;
195}

199void MPU_readProcessedData(I2C_HandleTypeDef *I2Cx)
200{
201 // Get raw values from the IMU
202 MPU_readRawData(I2Cx);
203
204 // Convert accelerometer values to g's
205 sensorData.ax = rawData.ax / aScaleFactor;
206 sensorData.ay = rawData.ay / aScaleFactor;
207 sensorData.az = rawData.az / aScaleFactor;
208
209 // Compensate for gyro offset
210 sensorData.gx = rawData.gx - gyroCal.x;
211 sensorData.gy = rawData.gy - gyroCal.y;
212 sensorData.gz = rawData.gz - gyroCal.z;
213
214 // Convert gyro values to deg/s
215 sensorData.gx /= gScaleFactor;
216 sensorData.gy /= gScaleFactor;
217 sensorData.gz /= gScaleFactor;
218}

222void MPU_calcAttitude(I2C_HandleTypeDef *I2Cx)
223{
224 // Read processed data
225 MPU_readProcessedData(I2Cx);
226
227 // Complementary filter
228 float accelPitch = atan2(sensorData.ay, sensorData.az) * RAD2DEG;
229 float accelRoll = atan2(sensorData.ax, sensorData.az) * RAD2DEG;
230
231 attitude.r = _tau * (attitude.r - sensorData.gy * _dt) + (1 - _tau) * accelRoll;
232 attitude.p = _tau * (attitude.p + sensorData.gx * _dt) + (1 - _tau) * accelPitch;
233 attitude.y += sensorData.gz * _dt;
234}
