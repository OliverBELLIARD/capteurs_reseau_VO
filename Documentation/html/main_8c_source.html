144void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim)
145{
146 if (htim->Instance == TIM2) // Check if the interrupt is from Timer 2
147 {
148 // Motor movement proportional to the temperature
149 angle += (int)(BMP280_get_temperature() * proportional_coeff)%360 - 180;
150
151 // Sign and angle change to do complete rotations
152 if (angle>0) MOT_Rotate(angle, MOT_ANGLE_POSITIVE);
153 else if (angle<0) MOT_Rotate(-angle, MOT_ANGLE_NEGATIVE);
154 }
155}

162void parse_RaspberryPI_Request(char* cmd)
163{
164 if (!strncmp(cmd, "GET_T", 5)) {
165 printf("T=+%.2f_C\r\n", (float) (BMP280_get_temperature()/100));
166 }
167 else if (!strncmp(cmd, "GET_P", 5)) {
168 printf("P=%dPa\r\n", (int) (BMP280_get_pressure()/256));
169 }
170 else if (!strncmp(cmd, "SET_K", 5)) {
171 MPU_calibrateGyro(&hi2c1, gyro_K);
172 printf("SET_K=OK\r\n");
173 }
174 else if (!strncmp(cmd, "GET_K", 5)) {
175 printf("K=%.5f\r\n", (float) gyro_K/100);
176 }
177 else if (!strncmp(cmd, "GET_A", 5)) {
178 MPU_calcAttitude(&hi2c1);
179 printf("A=%.1f;%.1f;%.1f\r\n", attitude.r, attitude.p, attitude.y);
180 }
181 else {
182 printf("Unknown request: %s\r\n", cmd);
183 }
184}
